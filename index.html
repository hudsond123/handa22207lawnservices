        // Render dashboard database listings

        function renderBookingsDashboard() {

            const feedContainer = document.getElementById('bookingsFeed');

            const searchVal = document.getElementById('bookingSearchInput').value.toLowerCase().replace(/\s/g, '');

            feedContainer.innerHTML = '';


            // Filter active bookings matching search (name, phone, or service package name)

            const filteredBookings = bookingsDB.filter(b => {

                const cleanPhone = b.phone.replace(/\D/g, "");

                const cleanName = b.name.toLowerCase();

                const cleanService = b.service.toLowerCase();

                return cleanPhone.includes(searchVal) || cleanName.includes(searchVal) || cleanService.includes(searchVal);

            });


            if (filteredBookings.length === 0) {

                feedContainer.innerHTML = `

                    <div class="text-center py-8 text-slate-400 text-xs sm:text-sm">

                        <i data-lucide="folder-open" class="mx-auto w-8 h-8 mb-2 opacity-55"></i>

                        No active reservations matching details. Try searching by client name, phone number, or service package.

                    </div>

                `;

                lucide.createIcons();

                return;

            }


            filteredBookings.forEach(b => {

                const bDateStr = new Date(b.date.replace(/-/g, '/')).toLocaleDateString('en-US', {

                    month: 'short', day: 'numeric', year: 'numeric'

                });

                

                const card = document.createElement('div');

                card.className = "flex flex-col sm:flex-row justify-between items-start sm:items-center bg-slate-50 hover:bg-slate-100/70 border border-slate-200/80 p-4 rounded-xl gap-4 transition-all";

                card.innerHTML = `

                    <div class="space-y-1">

                        <div class="flex items-center gap-2">

                            <span class="text-xs font-black text-slate-800">${b.name}</span>

                            <span class="text-[9px] font-bold text-slate-400 bg-white border px-1.5 py-0.5 rounded-md uppercase tracking-wider">${b.id}</span>

                        </div>

                        <p class="text-xs text-slate-500 font-medium">Service: <strong class="text-brand-700">${b.service}</strong></p>

                        <p class="text-[11px] text-slate-400">Date slot: <strong class="text-slate-600">${bDateStr} - ${b.timeslot}</strong></p>

                        <p class="text-[11px] text-slate-400">Phone: <strong class="text-slate-600">${b.phone}</strong></p>

                    </div>

                    <div class="flex items-center gap-3 w-full sm:w-auto justify-between sm:justify-end border-t sm:border-0 pt-3 sm:pt-0">

                        <span class="text-sm font-black text-slate-900">${b.price}</span>

                        <button type="button" onclick="cancelBooking('${b.id}')" class="text-xs text-red-500 hover:text-white hover:bg-red-500 border border-red-200 hover:border-red-500 font-bold px-3 py-1.5 rounded-lg transition-all flex items-center gap-1">

                            <i data-lucide="trash-2" class="w-3.5 h-3.5"></i> Cancel

                        </button>

            });

            lucide.createIcons();

        }


        // Cancel simulated booking slot

        function cancelBooking(id) {

            const index = bookingsDB.findIndex(b => b.id === id);

            if (index > -1) {

                bookingsDB.splice(index, 1);

                

                // Update Storage

                localStorage.setItem('handa_bookings', JSON.stringify(bookingsDB));


                // Refresh views

                renderCalendar();

                renderBookingsDashboard();

                

                showCustomAlert(

                    "trash-2",

                    "Reservation Canceled",

                    `Booking slot <strong>${id}</strong> has been officially removed from Hudson and Aaron's schedule.`,

                    "bg-red-50 text-red-700 border border-red-100"

                );

            }

        }
